122 T* Component::getComponent(Component* requester) {
123  // jump up in hierarchy to parent.
124  if (requester == nullptr) {
125  if (_parent != nullptr) {
126  // request from the parent to find a sibling.
127  // keeping track of the original requester.
128  return _parent->getComponent<T>(this);
129  }
130  else {
131  // no parent means no siblings.
132  // use get subcomponents if you want that.
133  return nullptr;
134  }
135  }
136 
137  // traverse children
138  for (auto c : getChildren()) {
139  if (c == requester) {
140  // skip original requester to avoid infinite recursions.
141  continue;
142  }
143 
144  T* t = dynamic_cast<T*>(c);
145  if (t != nullptr) {
146  return t;
147  }
148  }
149 
150  // jump up in hierarchy one stack frame deeper
151  if (_parent != nullptr) {
152  return _parent->getComponent<T>(this);
153  }
154 
155  return nullptr;
156 }
