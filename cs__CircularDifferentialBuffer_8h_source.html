208 _firstVal += _array[_head];

222 bool push(T value) {
223 if (!_contentSize) {
224 _firstVal = value;
225 _lastVal = value;
226 _contentSize=1;
227 return true;
228 }
229
230 int32_t diff = (int32_t)value - _lastVal;
231 if (diff > 127 || diff < -127) {
232// LOGd("difference too large! %i", diff);
233 clear();
234 return false;
235 }
236
237 incTail();
238 if (_contentSize > _capacity) {
239 _firstVal += _array[_head];
240 incHead();
241 }
242 _array[_tail] = diff;
243 _lastVal = value;
244 return true;
245 }
