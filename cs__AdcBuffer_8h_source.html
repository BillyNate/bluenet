57  cs_ret_code_t init() {
58  if (_allocated) {
59  return ERR_SUCCESS;
60  }
61  for (adc_buffer_id_t i = 0; i < ADC_BUFFER_COUNT; ++i) {
62  adc_buffer_t* buf = (adc_buffer_t*) malloc(sizeof(adc_buffer_t));
63  if (buf == nullptr) {
64  LOGw("No space to allocate buf");
65  // TODO: free all buffers?
66  return ERR_NO_SPACE;
67  }
68  buf->samples = (adc_sample_value_t*) malloc(sizeof(adc_sample_value_t) * ADC_BUFFER_SAMPLE_COUNT);
69  if (buf->samples == nullptr) {
70  LOGw("No space to allocate samples buf");
71  // TODO: free all buffers?
72  return ERR_NO_SPACE;
73  }
74  LOGd("Allocate buffer %i = %p", i, buf->samples);
75  _buf[i] = buf;
76  }
77  _allocated = true;
78  return ERR_SUCCESS;
79  }
