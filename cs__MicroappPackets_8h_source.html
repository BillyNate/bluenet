50struct __attribute__((__packed__)) microapp_binary_header_t {
51 uint8_t sdkVersionMajor; // Similar to microapp_sdk_version_t
52 uint8_t sdkVersionMinor;
53 uint16_t size; // Size of the binary, including this header.
54
55 uint16_t checksum; // Checksum (CRC16-CCITT) of the binary, after this header.
56 uint16_t checksumHeader; // Checksum (CRC16-CCITT) of this header, with this field set to 0.
57
58 uint32_t appBuildVersion; // Build version of this microapp.
59
60 uint16_t startOffset; // Offset in bytes of the first instruction to execute.
61 uint16_t reserved; // Reserved for future use, must be 0 for now.
62
63 uint32_t reserved2; // Reserved for future use, must be 0 for now.
64};

104struct __attribute__((packed)) microapp_state_t {
105 uint16_t checksum; // Checksum of the microapp, should be equal to the checksum field of the binary.
106 uint16_t checksumHeader; // Checksum of the microapp, should be equal to the checksumHeader field of the binary.
107
108 bool hasData: 1; // Whether the storage space of this app contains data.
109 uint8_t checksumTest: 2; // values: MICROAPP_TEST_STATE
110 bool enabled: 1; // Whether the microapp is enabled.
111 uint8_t bootTest: 2; // Values: MICROAPP_TEST_STATE. Checks if the microapp starts, registers callback function in IPC, and returns to firmware.
112 uint8_t memoryUsage: 1; // values: ok, excessive
113 uint8_t didReboot: 1;
114 uint16_t reservedTest: 8; // Reserved, must be 0 for now.
115
116 uint8_t tryingFunction = MICROAPP_FUNCTION_NONE; // Index of registered function that didn't pass yet, and that we are calling now. MICROAPP_FUNCTION_NONE for none.
117 uint8_t failedFunction = MICROAPP_FUNCTION_NONE; // Index of registered function that was tried, but didn't pass. MICROAPP_FUNCTION_NONE for none.
118 uint32_t passedFunctions; // Bitmask of registered functions that were called and returned to firmware successfully.
119};
