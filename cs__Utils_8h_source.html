148 inline static cs_ret_code_t findAdvType(uint8_t type, uint8_t* advData, uint8_t advLen, cs_data_t* foundData) {
149  int index = 0;
150  foundData->data = nullptr;
151  foundData->len = 0;
152  while (index < advLen-1) {
153  uint8_t fieldLen = advData[index];
154  uint8_t fieldType = advData[index + 1];
155  // Check if length is not 0 or larger than remaining advertisement data.
156  if (fieldLen == 0 || index + 1 + fieldLen > advLen) {
157  return ERR_NOT_FOUND;
158  }
159 
160  if (fieldType == type) {
161  foundData->data = &advData[index+2];
162  foundData->len = fieldLen-1;
163  return ERR_SUCCESS;
164  }
165  index += fieldLen+1;
166  }
167  return ERR_NOT_FOUND;
168 }
