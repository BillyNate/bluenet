18 constexpr int parseHexChar(const char chr) {
19  if ('0' <= chr && chr <= '9') {
20  return chr - '0';
21  }
22  else if ('a' <= chr && chr <= 'f') {
23  return chr - 'a' + 10;
24  }
25  else if ('A' <= chr && chr <= 'F') {
26  return chr - 'A' + 10;
27  }
28  return 0;
29 }

34 constexpr int parseHexPair(const char *str) {
