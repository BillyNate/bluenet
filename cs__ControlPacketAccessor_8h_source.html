103 cs_ret_code_t setPayload(buffer_ptr_t payload, cs_buffer_size_t size) {
104 checkInitialized();
105 if (size > PAYLOAD_SIZE) {
106 LOGe(STR_ERR_BUFFER_NOT_LARGE_ENOUGH);
107 return ERR_BUFFER_TOO_SMALL;
108 }
109 if (size != 0 && payload == nullptr) {
110 LOGe("Null pointer");
111 return ERR_BUFFER_UNASSIGNED;
112 }
113 _buffer->header.payloadSize = size;
114 if (size && _buffer->payload != payload) {
115 memmove(_buffer->payload, payload, size);
116 }
117 else {
118 LOGd("Skip copy");
119 }
120 return ERR_SUCCESS;
121 }

126 LOGd(FMT_ASSIGN_BUFFER_LEN, buffer, size);
127 _buffer = (control_packet_t<PAYLOAD_SIZE>*)buffer;
128 return ERR_SUCCESS;
