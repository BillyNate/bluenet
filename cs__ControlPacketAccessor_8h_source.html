101  cs_ret_code_t setPayload(buffer_ptr_t payload, cs_buffer_size_t size) {
102  checkInitialized();
103  if (size > PAYLOAD_SIZE) {
104  LOGe(STR_ERR_BUFFER_NOT_LARGE_ENOUGH);
105  return ERR_BUFFER_TOO_SMALL;
106  }
107  if (size != 0 && payload == nullptr) {
108  LOGe("Null pointer");
109  return ERR_BUFFER_UNASSIGNED;
110  }
111  _buffer->header.payloadSize = size;
112  if (size && _buffer->payload != payload) {
113  memmove(_buffer->payload, payload, size);
114  }
115  else {
116  LOGd("Skip copy");
117  }
118  return ERR_SUCCESS;
119  }
