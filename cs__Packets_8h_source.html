612  }
613  if (packet->count > LEGACY_MULTI_SWITCH_MAX_ITEM_COUNT) {
614  return false;
615  }
616  if (size < LEGACY_MULTI_SWITCH_HEADER_SIZE + packet->count * sizeof(cs_legacy_multi_switch_item_t)) {
617  return false;
618  }
619  return true;
