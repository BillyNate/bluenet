563inline bool cs_legacy_multi_switch_is_valid(const cs_legacy_multi_switch_t* packet, size16_t size) {
564 if (size < LEGACY_MULTI_SWITCH_HEADER_SIZE) {
565 return false;
566 }
567 if (packet->type != 0) {
568 return false;
569 }
570 if (packet->count > LEGACY_MULTI_SWITCH_MAX_ITEM_COUNT) {
571 return false;
572 }
573 if (size < LEGACY_MULTI_SWITCH_HEADER_SIZE + packet->count * sizeof(cs_legacy_multi_switch_item_t)) {
574 return false;
575 }
576 return true;
577}
