504  void written(uint16_t len) {
505  // We can flood the chip with writes and a potential forced disconnect will be delayed and could crash the chip.
506  //TODO: have this from the stack directly.
507  if (ConnectionEncryption::getInstance().allowedToWrite()) {
508  LOGi("Not allowed to write, disconnect in progress");
509  return;
510  }
511 
512  setGattValueLength(len);
513 
514  EncryptionAccessLevel accessLevel = NOT_SET;
515  // when using encryption, the packet needs to be decrypted
516  if (_status.aesEncrypted && _minAccessLevel < ENCRYPTION_DISABLED) {
517 
518  // the arithmetic type- and the string type characteristics have their hardcoded length in the valueLength
519  // the getValueLength() for those two types will always be the same and the setValueLength does nothing there.
520  // In the case of a characteristic with a dynamic buffer length we need to set the length ourselves.
521  // To do this we assume the length of the data is the same as the encrypted buffer minus the overhead for the encryption.
522  // The result can be zero padded but generally the payload has it's own length indication.
523  uint16_t decryptionBufferLength = ConnectionEncryption::getPlaintextBufferSize(getGattValueLength(), ConnectionEncryptionType::CTR);
524  setValueLength(decryptionBufferLength);
525 
526  cs_ret_code_t retCode = ConnectionEncryption::getInstance().decrypt(
527  cs_data_t(getGattValuePtr(), getGattValueLength()),
528  cs_data_t(getValuePtr(), getValueLength()),
529  accessLevel,
530  ConnectionEncryptionType::CTR
531  );
532 
533 
534  if (retCode != ERR_SUCCESS) {
535  LOGi("failed to decrypt retCode=%u", retCode);
536  ConnectionEncryption::getInstance().disconnect();
537  return;
538  }
539 
540  // disconnect on failure or if the user is not authenticated
541  if (!KeysAndAccess::getInstance().allowAccess(_minAccessLevel , accessLevel)) {
542  LOGi("insufficient access");
543  ConnectionEncryption::getInstance().disconnect();
544  return;
545  }
546  }
547  else {
548  accessLevel = ENCRYPTION_DISABLED;
549  setValueLength(len);
550  }
551 
552 
553 
554  LOGd("%s: onWrite()", _name);
555 
556  _callbackOnWrite(accessLevel, getValue(), getValueLength());
557  }
