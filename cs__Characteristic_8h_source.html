500 void written(uint16_t len) {
501 // We can flood the chip with writes and a potential forced disconnect will be delayed and could crash the chip.
502 //TODO: have this from the stack directly.
503 if (ConnectionEncryption::getInstance().allowedToWrite()) {
504 LOGi("Not allowed to write, disconnect in progress");
505 return;
506 }
507
508 setGattValueLength(len);
509
510 EncryptionAccessLevel accessLevel = NOT_SET;
511 // when using encryption, the packet needs to be decrypted
512 if (_status.aesEncrypted && _minAccessLevel < ENCRYPTION_DISABLED) {
513
514 // the arithmetic type- and the string type characteristics have their hardcoded length in the valueLength
515 // the getValueLength() for those two types will always be the same and the setValueLength does nothing there.
516 // In the case of a characteristic with a dynamic buffer length we need to set the length ourselves.
517 // To do this we assume the length of the data is the same as the encrypted buffer minus the overhead for the encryption.
518 // The result can be zero padded but generally the payload has it's own length indication.
519 uint16_t decryptionBufferLength = ConnectionEncryption::getPlaintextBufferSize(getGattValueLength(), ConnectionEncryptionType::CTR);
520 setValueLength(decryptionBufferLength);
521
522 cs_ret_code_t retCode = ConnectionEncryption::getInstance().decrypt(
523 cs_data_t(getGattValuePtr(), getGattValueLength()),
524 cs_data_t(getValuePtr(), getValueLength()),
525 accessLevel,
526 ConnectionEncryptionType::CTR
527 );
528
529
530 if (retCode != ERR_SUCCESS) {
531 LOGi("failed to decrypt retCode=%u", retCode);
532 ConnectionEncryption::getInstance().disconnect();
533 return;
534 }
535
536 // disconnect on failure or if the user is not authenticated
537 if (!KeysAndAccess::getInstance().allowAccess(_minAccessLevel , accessLevel)) {
538 LOGi("insufficient access");
539 ConnectionEncryption::getInstance().disconnect();
540 return;
541 }
542 }
543 else {
544 accessLevel = ENCRYPTION_DISABLED;
545 setValueLength(len);
546 }
547
548
549
550 LOGd("%s: onWrite()", _name);
551
552 _callbackOnWrite(accessLevel, getValue(), getValueLength());
553 }
