22constexpr const char* operationModeName(OperationMode const & mode) {
23 switch(mode) {
24 case OperationMode::OPERATION_MODE_SETUP:
25 return "SETUP";
26 case OperationMode::OPERATION_MODE_DFU:
27 return "DFU";
28 case OperationMode::OPERATION_MODE_FACTORY_RESET:
29 return "RESET";
30 case OperationMode::OPERATION_MODE_NORMAL:
31 return "NORMAL";
32 case OperationMode::OPERATION_MODE_UNINITIALIZED:
33 return "UNINIT";
34 }
35 // should never be reached
36 return "UNKNOWN";
37}

39constexpr int ValidMode(OperationMode const & mode) {
40 switch(mode) {
41 case OperationMode::OPERATION_MODE_SETUP:
42 case OperationMode::OPERATION_MODE_DFU:
43 case OperationMode::OPERATION_MODE_FACTORY_RESET:
44 case OperationMode::OPERATION_MODE_NORMAL:
45 case OperationMode::OPERATION_MODE_UNINITIALIZED:
46 return 1;
47 }
48 return 0;
49}
