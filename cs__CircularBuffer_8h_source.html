49 bool init() {
50 if (_array != nullptr) {
51 return false;
52 }
53 // Allocate memory
54 LOGCircularBufferDebug(FMT_ALLOCATE_MEMORY, _array);
55 _array = (T*)calloc(_capacity, sizeof(T));
56 if (_array == nullptr) {
57 LOGw(STR_ERR_ALLOCATE_MEMORY);
58 return false;
59 }
60
61 _allocatedSelf = true;
62 // Also call clear to make sure we start with a clean buffer
63 clear();
64 return true;
65 }
66
70 bool deinit() {
71 if (_array != nullptr && _allocatedSelf) {
72 free(_array);
73 }
74 _allocatedSelf = false;
75 _array = nullptr;
76 return true;
77 }

85 bool assign(buffer_ptr_t buffer, uint16_t bufferSize) {

93 // Also call clear to make sure we start with a clean buffer
