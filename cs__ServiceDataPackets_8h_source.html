190constexpr stone_id_t getStoneId(service_data_encrypted_t* data) {
191 switch (data->type) {
192 case SERVICE_DATA_DATA_TYPE_STATE: return data->state.id;
193 case SERVICE_DATA_DATA_TYPE_ERROR: return data->error.id;
194 case SERVICE_DATA_DATA_TYPE_EXT_STATE: return data->extState.id;
195 case SERVICE_DATA_DATA_TYPE_EXT_ERROR: return data->extError.id;
196 case SERVICE_DATA_DATA_TYPE_ALTERNATIVE_STATE: return data->altState.id;
197 case SERVICE_DATA_DATA_TYPE_HUB_STATE: return data->hubState.id;
198 case SERVICE_DATA_DATA_TYPE_MICROAPP: return data->microapp.id;
199 }
200 return 0;
201}
