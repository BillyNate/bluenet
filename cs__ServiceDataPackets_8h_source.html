185constexpr stone_id_t getStoneId(service_data_encrypted_t* data) {
186 switch (data->type) {
187 case SERVICE_DATA_DATA_TYPE_STATE: return data->state.id;
188 case SERVICE_DATA_DATA_TYPE_ERROR: return data->error.id;
189 case SERVICE_DATA_DATA_TYPE_EXT_STATE: return data->extState.id;
190 case SERVICE_DATA_DATA_TYPE_EXT_ERROR: return data->extError.id;
191 case SERVICE_DATA_DATA_TYPE_ALTERNATIVE_STATE: return data->altState.id;
192 case SERVICE_DATA_DATA_TYPE_HUB_STATE: return data->hubState.id;
193 case SERVICE_DATA_DATA_TYPE_MICROAPP: return data->microapp.id;
194 }
195 return 0;
196}
