16 inline int cs_write_test(const char *str, ...) {
17  char buffer[128];
18  va_list ap;
19  va_start(ap, str);
20  int16_t len = vsnprintf(NULL, 0, str, ap);
21  va_end(ap);
22 
23  if (len < 0) return len;
24 
25  // if strings are small we do not need to allocate by malloc
26  if (sizeof buffer >= len + 1UL) {
27  va_start(ap, str);
28  len = vsprintf(buffer, str, ap);
29  va_end(ap);
30  UartHandler::getInstance().writeMsg(UART_OPCODE_TX_FIRMWARESTATE, (uint8_t*)buffer, len);
31  } else {
32  char *p_buf = (char*)malloc(len + 1);
33  if (!p_buf) return -1;
34  va_start(ap, str);
35  len = vsprintf(p_buf, str, ap);
36  va_end(ap);
37  UartHandler::getInstance().writeMsg(UART_OPCODE_TX_FIRMWARESTATE, (uint8_t*)p_buf, len);
38  free(p_buf);
39  }
40  return len;
41 }

43 // when debug is not defined, define the centralized method
44 // as empty symbol so that all calls to it result in no generated code.

46 #endif
