15 uint8_t inline compressChannel(uint8_t channel) {
16  switch (channel) {
17  case 37: return 1;
18  case 38: return 2;
19  case 39: return 3;
20  default: return 0;
21  }
22 }

24 uint8_t inline decompressChannel(uint8_t compressedChannel) {
25  return compressedChannel == 0 ? 0 : compressedChannel + 36;
26 }

28 uint8_t inline compressRssi(int8_t rssi) {
29  if (rssi < 0) {
30  rssi = -rssi;
31  }
32  else {
33  // All positive values are compressed to 0.
34  rssi = 0;
35  }
36 
37  return rssi / 2;
38 }
39 
40 int8_t inline decompressRssi(uint8_t compressedRssi) {
41  return -2 * compressedRssi;
42 }

46 class rssi_and_channel_float_t {
47 public:
48  float _rssi; // signed rssi, usually negative. higher is closer.
49  uint8_t _channel; // 37-39
50 
51  rssi_and_channel_float_t(float rssi, uint8_t channel = 0) :_rssi(rssi), _channel(channel) {}
52 
53  bool isCloserThan(const rssi_and_channel_float_t& other) {
54  return _rssi > other._rssi;
55  }
56 
57  bool isCloserEqual(const rssi_and_channel_float_t & other) {
58  return isCloserThan(other) || _rssi == other._rssi;
59  }

68 struct __attribute__((__packed__)) rssi_and_channel_t {
69  uint8_t channel : 2; // 0 = unknown, 1 = channel 37, 2 = channel 38, 3 = channel 39
70  uint8_t rssiHalved : 6; // half of the absolute value of the original rssi.
71 
72  rssi_and_channel_t() = default;
73 
74  rssi_and_channel_t(int8_t rssi, uint8_t ch = 0) {
75  channel = compressChannel(ch);
76  rssiHalved = compressRssi(rssi);
77  }
78 
79  uint8_t getChannel() const {
80  return decompressChannel(channel);
81  }
82 
83  int8_t getRssi() const {
84  return decompressRssi(rssiHalved);
85  }
86 
87  rssi_and_channel_float_t toFloat() {
88  return rssi_and_channel_float_t (getRssi(), getChannel());
89  }

91  rssi_and_channel_float_t fallOff(float rate_db_sec, float dt_sec) {
