23  template <class T, class S>
24  auto mod(T v, S m) -> decltype(v%m) {
25  if(m == 0){
26  return v;
27  }
28  if(m < 0){
29  // flip [m] doesn't change the value.
30  return mod(v,-m);
31  }
32  if(v < 0){
33  // flip [v] twice won't change it either, but be careful not to return
34  // [m] instead of 0 when val is a multiple of m.
35  if(auto neg_v = mod(-v,m)){
36  // hidden recursion in if clause will only go one level deep ;)
37  return m - neg_v;
38  } else {
39  return 0;
40  }
41  }
42  // m positive, v non-negative, yay C++ actually works in that case.
43  return v%m;
44  }

92  // Interval<uint8_t> i(200,100)

94  // considers this interval as half open (lower closed) so that the return value is
95  // true if [val] is enclosed in the interval or equal to the lower boundary.
96  bool contains(T val){
97  return
98  low < high // reversed interval?
99  ? (low <= val && val < high) // nope, both must hold
100  : (low <= val || val < high); // yup, only one can hold
101  }
